基本工具包安装

sudo pacman-mirrors --country China    # 换源
sudo pacman -Syu    # 系统更新，manjaro是滚动更新方式，第一次更新很大，但一定要更新

sudo pacman -Sy neovim    # neovim，vim的新版
reboot    # 重启树莓派

sudo pacman -Sy base-devel # gcc等
sudo pacman -Sy gdb
sudo pacman -Sy glibc
sudo pacman -Sy cmake

# llvm 系
sudo pacman -S clang
sudo pacman -S lldb

sudo pacman -S code

  • 比起使用wpa_supplicant，manjaro可以使用nmcli（network manager）管理有线和无线网络；

或者使用nmtui，使用命令行下的界面；

sudo pacman -Syu netctl # 安装netctl
sudo cp /etc/netctl/examples/wireless-wpa /etc/netctl/myConfig  # 依据样例创建一个配置脚本文件
sudo nano /etc/netctl/myConfig # 按格式写要连的wifi名和密码
# 确保安装了wpa_actiond，如果没有安装使用 yay -S wpa_actiond
# 执行上面这条命令需要安装基本的编译构建环境和yay

sudo systemctl enable netctl-auto@wlan0.service # 将自动连接wifi启动为服务
